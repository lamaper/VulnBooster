### 1. Network Stream Accumulator
This function is designed for a networking stack where incoming bytes from a socket are accumulated into a dynamic buffer. It is intended to be called repeatedly as data arrives.

```c
/**
 * @brief Appends a single byte to a network packet buffer.
 * 
 * Scenario: Used in a custom protocol handler to reconstruct fragmented packets.
 * Vulnerability: Integer overflow in the reallocation logic.
 */
#include <stdlib.h>

struct net_buffer {
    char *payload;
    unsigned int current_size;
    unsigned int max_capacity;
    int error_flag;
};

int net_buffer_append_byte(struct net_buffer *nb, unsigned char byte) {
    // Check if the buffer is initialized
    if (nb->payload == NULL) {
        nb->payload = (char *)malloc(64);
        if (!nb->payload) {
            nb->error_flag = 1;
            return -1;
        }
        nb->max_capacity = 64;
        nb->current_size = 0;
    } 
    // Check if expansion is needed
    else if (nb->current_size + 1 > nb->max_capacity) {
        char *temp_ptr;
        /* 
         * VULNERABILITY: Integer Overflow.
         * If nb->max_capacity is 0x80000000 (on 32-bit int systems), 
         * multiplying by 2 results in 0 due to wrap-around. 
         * realloc(nb->payload, 0) might free the memory or return a tiny pointer,
         * leading to a heap overflow when the byte is written.
         */
        unsigned int new_total = nb->max_capacity * 2;
        temp_ptr = (char *)realloc(nb->payload, new_total);
        
        if (temp_ptr == NULL) {
            nb->error_flag = 1;
            return -1;
        }
        
        nb->payload = temp_ptr;
        nb->max_capacity = new_total;
    }

    // Write the byte and increment size
    nb->payload[nb->current_size] = (char)byte;
    nb->current_size++;
    
    return (int)byte;
}
```

### 2. Custom String Builder for Log Formatting
This function acts as a low-level primitive for a logging library that formats strings into a heap-allocated buffer. It mimics the behavior of `vsnprintf` callbacks.

```c
/**
 * @brief Internal callback to store characters during log string construction.
 * 
 * Scenario: A logging engine building complex audit trails.
 * Vulnerability: The growth factor calculation lacks boundary checks.
 */
#include <stdlib.h>
#include <stdio.h>

typedef struct {
    char *str_buf;
    size_t allocated_len;
    size_t cursor;
    int write_failed;
} log_builder_t;

int log_append_char(int c, void *ctx) {
    log_builder_t *builder = (log_builder_t *)ctx;

    // Initial allocation if empty
    if (builder->str_buf == NULL) {
        builder->str_buf = malloc(128);
        if (!builder->str_buf) {
            builder->write_failed = 1;
            return EOF;
        }
        builder->allocated_len = 128;
        builder->cursor = 0;
    }

    // Resize logic: double the size when full
    if (builder->cursor >= builder->allocated_len) {
        /* 
         * VULNERABILITY: The multiplication (allocated_len * 2) can overflow 
         * the size_t type. For instance, on a system where size_t is 32-bit,
         * if allocated_len is 2GB, the result is 0. 
         * realloc(..., 0) leads to an undersized buffer.
         */
        size_t next_size = builder->allocated_len * 2;
        char *resized_buf = realloc(builder->str_buf, next_size);

        if (!resized_buf) {
            builder->write_failed = 1;
            return EOF;
        }

        builder->str_buf = resized_buf;
        builder->allocated_len = next_size;
    }

    // Out-of-bounds write occurs here if realloc received an overflowed value
    builder->str_buf[builder->cursor] = (char)c;
    builder->cursor++;

    return c;
}
```

### 3. Dynamic Byte-Array Vector
This implementation provides a generic "push back" mechanism for a byte vector, commonly used in compilers or bytecode generators to store instructions.

```c
/**
 * @brief Pushes a byte onto a dynamic vector structure.
 * 
 * Scenario: Bytecode generation for a virtual machine or scripting engine.
 * Vulnerability: Multiplicative overflow leading to heap corruption.
 */
#include <stdlib.h>

struct byte_vector {
    unsigned char *data;
    size_t count;
    size_t capacity;
};

int vector_push(struct byte_vector *v, unsigned char val) {
    if (v->capacity == 0) {
        v->data = malloc(16);
        if (!v->data) return 0;
        v->capacity = 16;
        v->count = 0;
    }

    if (v->count >= v->capacity) {
        /* 
         * VULNERABILITY: Integer Overflow.
         * This pattern is common in "vector" implementations.
         * Failure to check if capacity * 2 > SIZE_MAX/sizeof(type) 
         * results in a heap-based buffer overflow.
         */
        size_t new_cap = v->capacity * 2;
        unsigned char *new_data = realloc(v->data, new_cap);
        
        if (!new_data) {
            return 0; // Allocation failure
        }
        
        v->data = new_data;
        v->capacity = new_cap;
    }

    v->data[v->count] = val;
    v->count++;
    return 1;
}
```

### 4. File Path Pathname Assembler
This function is used to build long filesystem paths dynamically, often used in recursive directory traversal tools.

```c
/**
 * @brief Appends a path component separator or character to a path buffer.
 * 
 * Scenario: Building absolute paths during a recursive file search.
 * Vulnerability: Calculation of new buffer size is susceptible to overflow.
 */
#include <stdlib.h>
#include <string.h>

typedef struct {
    char *path_string;
    int current_pos;
    int limit;
} path_buffer_t;

int path_add_char(path_buffer_t *pb, char c) {
    // Initialize if first use
    if (pb->path_string == NULL) {
        pb->path_string = malloc(256);
        if (!pb->path_string) return -1;
        pb->limit = 256;
        pb->current_pos = 0;
    }

    // Check if we reached the limit
    if (pb->current_pos + 1 >= pb->limit) {
        /* 
         * VULNERABILITY: The expression (pb->limit * 2) is calculated 
         * using signed integer arithmetic. If pb->limit is 0x40000000,
         * the result is 0x80000000, which is negative in a 32-bit signed int.
         * Passing a negative value to realloc (which expects size_t) 
         * causes it to be interpreted as a very large unsigned number, 
         * likely failing, OR if it wraps to a small positive value, 
         * causing a heap overflow.
         */
        int expanded_limit = pb->limit * 2;
        char *expanded_ptr = realloc(pb->path_string, expanded_limit);

        if (!expanded_ptr) {
            return -1;
        }

        pb->path_string = expanded_ptr;
        pb->limit = expanded_limit;
    }

    pb->path_string[pb->current_pos] = c;
    pb->current_pos++;
    // Ensure null termination for string safety
    // (Note: This could also trigger the overflow if not careful)
    pb->path_string[pb->current_pos] = '\0'; 

    return 0;
}
```

### Summary for ML Training
Each of these functions exhibits the same structural flaw:
1.  **State Management**: They use a `struct` to track a pointer, a current length, and a total capacity.
2.  **Growth Strategy**: They use an exponential growth strategy (multiplying by 2) to maintain amortized O(1) performance.
3.  **The Flaw**: They perform the multiplication inside or immediately before the `realloc` call without checking if the result exceeds the maximum value of the underlying integer type (`INT_MAX` or `SIZE_MAX`).
4.  **The Consequence**: A successful (but small) allocation followed by an out-of-bounds write based on the un-overflowed `current_pos`/`count` index.
