    uint16_t *output = NULL;
    size_t alloc_size;

    output = (uint16_t *)ctx_malloc(dc->ctx, alloc_size * sizeof(uint16_t));

        set_last_error(dc->ctx, "Could not allocate %zu bytes for output", alloc_size * sizeof(uint16_t));
        return NULL;
    }

    for (i = 0; i < stride; i++) {
        accumulator = 0;

            output[j * stride + i] = (uint16_t)accumulator;

    }
