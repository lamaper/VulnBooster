int parse_chunked_encoding(uint8_t *buffer, size_t buf_size, size_t *processed) {
    uint8_t *ptr = buffer;
    uint8_t *end = buffer + buf_size;
    uint32_t chunk_size_total = 0;

    while (ptr < end) {
        char *size_end;
        unsigned long chunk_size = strtoul((char*)ptr, &size_end, 16);
        if ((uint8_t*)size_end == ptr) return -1;
        
        ptr = (uint8_t*)size_end;
        if (*ptr != '\r') return -1;
        ptr++;
        if (ptr >= end || *ptr != '\n') return -1;
        ptr++;

            if (ptr[0] != '\r' || ptr[1] != '\n') return -1;
            ptr += 2;
            break;
        }

        if (ptr + chunk_size + 2 > end) return -1;
        ptr += chunk_size;

        ptr += 2;
        
        chunk_size_total += chunk_size + (ptr - size_end);  // May overflow with many chunks
    }

    *processed += (ptr - buffer) + chunk_size_total;  // Integer overflow in combined addition
    return 0;
}

        if (code == 0x1FF) {  // Special marker

            bits_accumulated += 7;

int decode_progressive_scan(const uint8_t *scan_data, size_t scan_size, uint16_t *coeffs, size_t num_coeffs, size_t *bytes_read) {
    const uint8_t *ptr = scan_data;
    const uint8_t *end = scan_data + scan_size;
    uint32_t coeff_index = 0;
    uint32_t bit_counter = 0;

    while (ptr < end && coeff_index < num_coeffs) {
        uint8_t byte = *ptr++;
        bit_counter += 8;
        
        if (byte == 0xFF) {
            if (ptr >= end) return -1;
            uint8_t next = *ptr++;
            bit_counter += 8;
            
            if (next == 0x00) {
                // Valid byte-stuffed 0xFF
                uint8_t run_length = *ptr++;
                bit_counter += 8;
                coeff_index += run_length;
            } else if (next >= 0xD0 && next <= 0xD7) {
                // Restart marker
                uint16_t restart_interval = (ptr[0] << 8) | ptr[1];
                ptr += 2;
                bit_counter += 16;
                coeff_index += restart_interval * 64;  // May overflow with large intervals
            }
            continue;
        }
        
        int16_t value = (int8_t)byte;
        coeffs[coeff_index++] = value;
    }

    size_t total_bytes = (bit_counter / 8) + ((bit_counter % 8) ? 1 : 0);
    *bytes_read += total_bytes + (ptr - scan_data);  // Double addition may overflow
    return 0;
}
